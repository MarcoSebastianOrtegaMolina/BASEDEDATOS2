// Función auxiliar para no repetir Math.random tantas veces
const obtenerAleatorio = (arreglo) => arreglo[Math.floor(Math.random() * arreglo.length)];

function iniciarCreacion() {
    const formato = document.getElementById("formatoSalida").value;
    
    if (formato === "sql" || formato === "postgres") {
        crearSQL();
    } else if (formato === "csv") {
        crearCSV();
    } else if (formato === "json") {
        crearJSON();
    }
}

function crearSQL() {
    let matriculaBase = 224250000;
    let cantidad = parseInt(document.getElementById('numRegistros').value);
    let comandos = [];
    
    for (let i = 0; i < cantidad; i++) {
        let ap1 = obtenerAleatorio(apellidosUno);
        let ap2 = obtenerAleatorio(apellidosDos);
        let nom1 = obtenerAleatorio(nombresUno);
        
        // 50% de probabilidad de tener un segundo nombre
        let nom2 = Math.random() < 0.5 ? "" : ` ${obtenerAleatorio(nombresDos)}`;
        let nombreCompleto = nom1 + nom2;
        
        let ap2Sql = ap2 === "NULL" ? "NULL" : `UPPER('${ap2}')`;
        let correo = `a${matriculaBase + i}@unison.mx`;
        
        comandos.push(`(${matriculaBase + i}, UPPER('${ap1}'), ${ap2Sql}, '${nombreCompleto}', '${correo}')`);
    }
    
    datosGenerados = "INSERT INTO alumnos (matricula, apellido1, apellido2, nombre, correo) VALUES \n" + comandos.join(",\n") + ";";
    document.getElementById("areaResultado").innerHTML = datosGenerados.replace(/\n/g, "<br>");
}

function crearCSV() {
    let matriculaBase = 224250000;
    let cantidad = parseInt(document.getElementById('numRegistros').value);
    datosGenerados = "matricula,apellido1,apellido2,nombre,correo\n";
    
    for (let i = 0; i < cantidad; i++) {
        let ap1 = obtenerAleatorio(apellidosUno);
        let ap2 = obtenerAleatorio(apellidosDos);
        let nom = obtenerAleatorio(nombresUno) + (Math.random() < 0.5 ? ` ${obtenerAleatorio(nombresDos)}` : "");
        let correo = `a${matriculaBase + i}@unison.mx`;
        
        let ap2Csv = ap2 === "NULL" ? "" : ap2; // En CSV es mejor dejar vacío en vez de "NULL" textual
        
        datosGenerados += `${matriculaBase + i},${ap1},${ap2Csv},${nom},${correo}\n`;
    }
    
    document.getElementById("areaResultado").innerHTML = datosGenerados.replace(/\n/g, "<br>");
}

function crearJSON() {
    let matriculaBase = 224250000;
    let cantidad = parseInt(document.getElementById('numRegistros').value);
    let estudiantes = [];

    for (let i = 0; i < cantidad; i++) {
        let ap2 = obtenerAleatorio(apellidosDos);
        
        estudiantes.push({
            matricula: matriculaBase + i,
            apellido1: obtenerAleatorio(apellidosUno),
            apellido2: ap2 === "NULL" ? null : ap2,
            nombre: obtenerAleatorio(nombresUno) + (Math.random() < 0.5 ? ` ${obtenerAleatorio(nombresDos)}` : ""),
            correo: `a${matriculaBase + i}@unison.mx`
        });
    }
    
    // Convertimos el objeto nativo a formato JSON con indentación de 4 espacios
    datosGenerados = JSON.stringify(estudiantes, null, 4);
    
    // Para mostrarlo en el HTML respetando los espacios
    document.getElementById("areaResultado").innerHTML = datosGenerados.replace(/\n/g, "<br>").replace(/ /g, "&nbsp;");
}

function descargarArchivo() {
    if (!datosGenerados) {
        alert("Primero debes generar los datos.");
        return;
    }

    let formato = document.getElementById("formatoSalida").value;
    let extension = formato === "postgres" ? "sql" : formato;
    
    let enlace = document.createElement("a");
    enlace.setAttribute("href", "data:text/plain;charset=UTF-8," + encodeURIComponent(datosGenerados));
    enlace.setAttribute("download", `estudiantes_generados.${extension}`);
    
    document.body.appendChild(enlace);
    enlace.click();
    document.body.removeChild(enlace);
}
